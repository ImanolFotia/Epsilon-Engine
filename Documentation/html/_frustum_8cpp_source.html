43 void NormalizePlane(float frustum[6][4], int side)
44 {
45  // Here we calculate the magnitude of the normal to the plane (point A B C)
46  // Remember that (A, B, C) is that same thing as the normal's (X, Y, Z).
47  // To calculate magnitude you use the equation: magnitude = sqrt( x^2 + y^2 + z^2)
48  float magnitude = (float)sqrt( frustum[side][A] * frustum[side][A] +
49  frustum[side][B] * frustum[side][B] +
50  frustum[side][C] * frustum[side][C] );
51 
52  // Then we divide the plane's values by it's magnitude.
53  // This makes it easier to work with.
54  frustum[side][A] /= magnitude;
55  frustum[side][B] /= magnitude;
56  frustum[side][C] /= magnitude;
57  frustum[side][D] /= magnitude;
58 }

68 {

70  float modl[16]; // This will hold our modelview matrix
71  float clip[16]; // This will hold the clipping planes
72 
73  // glGetFloatv() is used to extract information about our OpenGL world.
74  // Below, we pass in GL_PROJECTION_MATRIX to abstract our projection matrix.
75  // It then stores the matrix into an array of [16].
76  for(int i = 0 ; i < 4 ; i++)
77  {
78  for(int j = 0 ; j < 4 ; j++)
79  {
80  proj[j+(4*i)] = ProjectionMatrix[i][j];
81  }
82  }
83  // By passing in GL_MODELVIEW_MATRIX, we can abstract our model view matrix.
84  // This also stores it in an array of [16].
85  for(int i = 0 ; i < 4 ; i++)
86  {
87  for(int j = 0 ; j < 4 ; j++)
88  {
89  modl[j+(4*i)] = ModelMatrix[i][j];
90  }
91  }
92 
93  // Now that we have our modelview and projection matrix, if we combine these 2 matrices,
94  // it will give us our clipping planes. To combine 2 matrices, we multiply them.
95 
96  clip[ 0] = modl[ 0] * proj[ 0] + modl[ 1] * proj[ 4] + modl[ 2] * proj[ 8] + modl[ 3] * proj[12];

117  // Now we actually want to get the sides of the frustum. To do this we take
118  // the clipping planes we received above and extract the sides from them.
119 
120  // This will extract the RIGHT side of the frustum
121  m_Frustum[RIGHT][A] = clip[ 3] - clip[ 0];
122  m_Frustum[RIGHT][B] = clip[ 7] - clip[ 4];
123  m_Frustum[RIGHT][C] = clip[11] - clip[ 8];
124  m_Frustum[RIGHT][D] = clip[15] - clip[12];

135  m_Frustum[LEFT][C] = clip[11] + clip[ 8];

142  m_Frustum[BOTTOM][A] = clip[ 3] + clip[ 1];
