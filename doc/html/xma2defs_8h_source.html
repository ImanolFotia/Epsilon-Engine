24 // A typical XMA2 file contains these RIFF chunks:
25 //
26 // 'fmt' or 'XMA2' chunk (or both): A description of the XMA data's structure
27 // and characteristics (length, channels, sample rate, loops, block size, etc).
28 //
29 // 'seek' chunk: A seek table to help navigate the XMA data.
30 //
31 // 'data' chunk: The encoded XMA2 data.
32 //
33 // The encoded XMA2 data is structured as a set of BLOCKS, which contain PACKETS,
34 // which contain FRAMES, which contain SUBFRAMES (roughly speaking). The frames
35 // in a file may also be divided into several subsets, called STREAMS.
36 //
37 // FRAME: A variable-sized segment of XMA data that decodes to exactly 512 mono
38 // or stereo PCM samples. This is the smallest unit of XMA data that can
39 // be decoded in isolation. Frames are an arbitrary number of bits in
40 // length, and need not be byte-aligned. See "XMA frame structure" below.
41 //
42 // SUBFRAME: A region of bits in an XMA frame that decodes to 128 mono or stereo
43 // samples. The XMA decoder cannot decode a subframe in isolation; it needs
44 // a whole frame to work with. However, it can begin emitting the frame's
45 // decoded samples at any one of the four subframe boundaries. Subframes
46 // can be addressed for seeking and looping purposes.
47 //
48 // PACKET: A 2Kb region containing a 32-bit header and some XMA frames. Frames
49 // can (and usually do) span packets. A packet's header includes the offset
50 // in bits of the first frame that begins within that packet. All of the
51 // frames that begin in a given packet belong to the same "stream" (see the
52 // Multichannel Audio section below).
53 //
54 // STREAM: A set of packets within an XMA file that all contain data for the
55 // same mono or stereo component of a PCM file with more than two channels.
56 // The packets comprising a given stream may be interleaved with each other
57 // more or less arbitrarily; see Multichannel Audio.
58 //
59 // BLOCK: An array of XMA packets; or, to break it down differently, a series of
60 // consecutive XMA frames, padded at the end with reserved data. A block
61 // must contain at least one 2Kb packet per stream, and it can hold up to
62 // 4095 packets (8190Kb), but its size is typically in the 32Kb-128Kb range.
63 // (The size chosen involves a trade-off between memory use and efficiency
64 // of reading from permanent storage.)
65 //
66 // XMA frames do not span blocks, so a block is guaranteed to begin with a
67 // set of complete frames, one per stream. Also, a block in a multi-stream
68 // XMA2 file always contains the same number of samples for each stream;
69 // see Multichannel Audio.
70 //
71 // The 'data' chunk in an XMA2 file is an array of XMA2WAVEFORMAT.BlockCount XMA
72 // blocks, all the same size (as specified in XMA2WAVEFORMAT.BlockSizeInBytes)
73 // except for the last one, which may be shorter.
74 
75 
76 // MULTICHANNEL AUDIO: the XMA decoder can only decode raw XMA data into either
77 // mono or stereo PCM data. In order to encode a 6-channel file (say), the file
78 // must be deinterleaved into 3 stereo streams that are encoded independently,
79 // producing 3 encoded XMA data streams. Then the packets in these 3 streams
80 // are interleaved to produce a single XMA2 file, and some information is added
81 // to the file so that the original 6-channel audio can be reconstructed at
82 // decode time. This works using the concept of an XMA stream (see above).
83 //
84 // The frames for all the streams in an XMA file are interleaved in an arbitrary
85 // order. To locate a frame that belongs to a given stream in a given XMA block,
86 // you must examine the first few packets in the block. Here (and only here) the
87 // packets are guaranteed to be presented in stream order, so that all frames
88 // beginning in packet 0 belong to stream 0 (the first stereo pair), etc.
89 //
90 // (This means that when decoding multi-stream XMA files, only entire XMA blocks
91 // should be submitted to the decoder; otherwise it cannot know which frames
92 // belong to which stream.)
93 //
94 // Once you have one frame that belongs to a given stream, you can find the next
95 // one by looking at the frame's 'NextFrameOffsetBits' value (which is stored in
96 // its first 15 bits; see XMAFRAME below). The GetXmaFrameBitPosition function
97 // uses this technique.
98 
99 
100 // SEEKING IN XMA2 FILES: Here is some pseudocode to find the byte position and
101 // subframe in an XMA2 file which will contain sample S when decoded.
102 //
103 // 1. Traverse the seek table to find the XMA2 block containing sample S. The
104 // seek table is an array of big-endian DWORDs, one per block in the file.
105 // The Nth DWORD is the total number of PCM samples that would be obtained
106 // by decoding the entire XMA file up to the end of block N. Hence, the
107 // block we want is the first one whose seek table entry is greater than S.
108 // (See the GetXmaBlockContainingSample helper function.)
109 //
110 // 2. Calculate which frame F within the block found above contains sample S.
111 // Since each frame decodes to 512 samples, this is straightforward. The
112 // first frame in the block produces samples X to X + 512, where X is the
113 // seek table entry for the prior block. So F is (S - X) / 512.
114 //
115 // 3. Find the bit offset within the block where frame F starts. Since frames
116 // are variable-sized, this can only be done by traversing all the frames in
117 // the block until we reach frame F. (See GetXmaFrameBitPosition.)
118 //
119 // 4. Frame F has four 128-sample subframes. To find the subframe containing S,
120 // we can use the formula (S % 512) / 128.
121 //
122 // In the case of multi-stream XMA files, sample S is a multichannel sample with
123 // parts coming from several frames, one per stream. To find all these frames,
124 // steps 2-4 need to be repeated for each stream N, using the knowledge that the
125 // first packets in a block are presented in stream order. The frame traversal
126 // in step 3 must be started at the first frame in the Nth packet of the block,
127 // which will be the first frame for stream N. (And the packet header will tell
128 // you the first frame's start position within the packet.)
129 //
130 // Step 1 can be performed using the GetXmaBlockContainingSample function below,
131 // and steps 2-4 by calling GetXmaDecodePositionForSample once for each stream.
132

333 // Helper functions to extract the fields above from an XMA packet. (Note that
334 // the bitfields cannot be read directly on little-endian architectures such as
335 // the Intel x86, as they are laid out in big-endian form.)

337 __inline DWORD GetXmaPacketFrameCount(__in_bcount(1) const BYTE* pPacket)
338 {
339  return (DWORD)(pPacket[0] >> 2);
340 }

342 __inline DWORD GetXmaPacketFirstFrameOffsetInBits(__in_bcount(3) const BYTE* pPacket)
343 {
344  return ((DWORD)(pPacket[0] & 0x3) << 13) |
345  ((DWORD)(pPacket[1]) << 5) |
346  ((DWORD)(pPacket[2]) >> 3);
347 }

351  return (DWORD)(pPacket[2] & 0x7);

398 // GetXmaBlockContainingSample: Use a given seek table to find the XMA block
399 // containing a given decoded sample. Note that the seek table entries in an
400 // XMA file are stored in big-endian form and may need to be converted prior
401 // to calling this function.
402 
403 __inline HRESULT GetXmaBlockContainingSample
404 (
405  DWORD nBlockCount, // Blocks in the file (= seek table entries)
406  __in_ecount(nBlockCount) const DWORD* pSeekTable, // Pointer to the seek table data
407  DWORD nDesiredSample, // Decoded sample to locate
408  __out DWORD* pnBlockContainingSample, // Index of the block containing the sample
409  __out DWORD* pnSampleOffsetWithinBlock // Position of the sample in this block
410 )
411 {
412  DWORD nPreviousTotalSamples = 0;
413  DWORD nBlock;
414  DWORD nTotalSamplesSoFar;
415 
416  XMA2DEFS_ASSERT(pSeekTable);
417  XMA2DEFS_ASSERT(pnBlockContainingSample);
418  XMA2DEFS_ASSERT(pnSampleOffsetWithinBlock);
419 
420  for (nBlock = 0; nBlock < nBlockCount; ++nBlock)
421  {
422  nTotalSamplesSoFar = pSeekTable[nBlock];
423  if (nTotalSamplesSoFar > nDesiredSample)
424  {
425  *pnBlockContainingSample = nBlock;
426  *pnSampleOffsetWithinBlock = nDesiredSample - nPreviousTotalSamples;
427  return S_OK;
428  }
429  nPreviousTotalSamples = nTotalSamplesSoFar;
430  }
431 
432  return E_FAIL;
433 }

436 // GetXmaFrameLengthInBits: Reads a given frame's LengthInBits field.
437 
438 __inline DWORD GetXmaFrameLengthInBits
439 (
440  __in_bcount(nBitPosition / 8 + 3)
441  __in const BYTE* pPacket, // Pointer to XMA packet[s] containing the frame
442  DWORD nBitPosition // Bit offset of the frame within this packet
443 )
444 {
445  DWORD nRegion;
446  DWORD nBytePosition = nBitPosition / 8;
447  DWORD nBitOffset = nBitPosition % 8;
448 
449  if (nBitOffset < 2) // Only need to read 2 bytes (and might not be safe to read more)
450  {
451  nRegion = (DWORD)(pPacket[nBytePosition+0]) << 8 |
452  (DWORD)(pPacket[nBytePosition+1]);
453  return (nRegion >> (1 - nBitOffset)) & 0x7FFF; // Last 15 bits
454  }
455  else // Need to read 3 bytes
456  {
457  nRegion = (DWORD)(pPacket[nBytePosition+0]) << 16 |
458  (DWORD)(pPacket[nBytePosition+1]) << 8 |
459  (DWORD)(pPacket[nBytePosition+2]);
460  return (nRegion >> (9 - nBitOffset)) & 0x7FFF; // Last 15 bits
461  }
462 }

465 // GetXmaFrameBitPosition: Calculates the bit offset of a given frame within
466 // an XMA block or set of blocks. Returns 0 on failure.
467 
468 __inline DWORD GetXmaFrameBitPosition
469 (
470  __in_bcount(nXmaDataBytes) const BYTE* pXmaData, // Pointer to XMA block[s]
471  DWORD nXmaDataBytes, // Size of pXmaData in bytes
472  DWORD nStreamIndex, // Stream within which to seek
473  DWORD nDesiredFrame // Frame sought
474 )
475 {
476  const BYTE* pCurrentPacket;
477  DWORD nPacketsExamined = 0;
478  DWORD nFrameCountSoFar = 0;
479  DWORD nFramesToSkip;
480  DWORD nFrameBitOffset;
481 
482  XMA2DEFS_ASSERT(pXmaData);
483  XMA2DEFS_ASSERT(nXmaDataBytes % XMA_BYTES_PER_PACKET == 0);
484 
485  // Get the first XMA packet belonging to the desired stream, relying on the
486  // fact that the first packets for each stream are in consecutive order at
487  // the beginning of an XMA block.
488 
489  pCurrentPacket = pXmaData + nStreamIndex * XMA_BYTES_PER_PACKET;
490  for (;;)
491  {
492  // If we have exceeded the size of the XMA data, return failure
493  if (pCurrentPacket + XMA_BYTES_PER_PACKET > pXmaData + nXmaDataBytes)
494  {
495  return 0;
496  }
497 
498  // If the current packet contains the frame we are looking for...
499  if (nFrameCountSoFar + GetXmaPacketFrameCount(pCurrentPacket) > nDesiredFrame)
500  {
501  // See how many frames in this packet we need to skip to get to it
502  XMA2DEFS_ASSERT(nDesiredFrame >= nFrameCountSoFar);
503  nFramesToSkip = nDesiredFrame - nFrameCountSoFar;
504 
505  // Get the bit offset of the first frame in this packet
506  nFrameBitOffset = XMA_PACKET_HEADER_BITS + GetXmaPacketFirstFrameOffsetInBits(pCurrentPacket);
507 
508  // Advance nFrameBitOffset to the frame of interest
509  while (nFramesToSkip--)
510  {
511  nFrameBitOffset += GetXmaFrameLengthInBits(pCurrentPacket, nFrameBitOffset);
512  }
513 
514  // The bit offset to return is the number of bits from pXmaData to
515  // pCurrentPacket plus the bit offset of the frame of interest
516  return (DWORD)(pCurrentPacket - pXmaData) * 8 + nFrameBitOffset;
517  }
518 
519  // If we haven't found the right packet yet, advance our counters
520  ++nPacketsExamined;
521  nFrameCountSoFar += GetXmaPacketFrameCount(pCurrentPacket);
522 
523  // And skip to the next packet belonging to the same stream
524  pCurrentPacket += XMA_BYTES_PER_PACKET * (GetXmaPacketSkipCount(pCurrentPacket) + 1);
525  }
526 }

529 // GetLastXmaFrameBitPosition: Calculates the bit offset of the last complete
530 // frame in an XMA block or set of blocks.
531 
532 __inline DWORD GetLastXmaFrameBitPosition
533 (
534  __in_bcount(nXmaDataBytes) const BYTE* pXmaData, // Pointer to XMA block[s]
535  DWORD nXmaDataBytes, // Size of pXmaData in bytes
536  DWORD nStreamIndex // Stream within which to seek
537 )
538 {
539  const BYTE* pLastPacket;
540  DWORD nBytesToNextPacket;
541  DWORD nFrameBitOffset;
542  DWORD nFramesInLastPacket;
543 
544  XMA2DEFS_ASSERT(pXmaData);
545  XMA2DEFS_ASSERT(nXmaDataBytes % XMA_BYTES_PER_PACKET == 0);
546  XMA2DEFS_ASSERT(nXmaDataBytes >= XMA_BYTES_PER_PACKET * (nStreamIndex + 1));
547 
548  // Get the first XMA packet belonging to the desired stream, relying on the
549  // fact that the first packets for each stream are in consecutive order at
550  // the beginning of an XMA block.
551  pLastPacket = pXmaData + nStreamIndex * XMA_BYTES_PER_PACKET;
552 
553  // Search for the last packet belonging to the desired stream
554  for (;;)
555  {
556  nBytesToNextPacket = XMA_BYTES_PER_PACKET * (GetXmaPacketSkipCount(pLastPacket) + 1);
557  XMA2DEFS_ASSERT(nBytesToNextPacket);
558  if (pLastPacket + nBytesToNextPacket + XMA_BYTES_PER_PACKET > pXmaData + nXmaDataBytes)
559  {
560  break; // The next packet would extend beyond the end of pXmaData
561  }
562  pLastPacket += nBytesToNextPacket;
563  }
564 
565  // The last packet can sometimes have no seekable frames, in which case we
566  // have to use the previous one
567  if (GetXmaPacketFrameCount(pLastPacket) == 0)
568  {
569  pLastPacket -= nBytesToNextPacket;
570  }
571 
572  // Found the last packet. Get the bit offset of its first frame.
573  nFrameBitOffset = XMA_PACKET_HEADER_BITS + GetXmaPacketFirstFrameOffsetInBits(pLastPacket);
574 
575  // Traverse frames until we reach the last one
576  nFramesInLastPacket = GetXmaPacketFrameCount(pLastPacket);
577  while (--nFramesInLastPacket)
578  {
579  nFrameBitOffset += GetXmaFrameLengthInBits(pLastPacket, nFrameBitOffset);
580  }
581 
582  // The bit offset to return is the number of bits from pXmaData to
583  // pLastPacket plus the offset of the last frame in this packet.
584  return (DWORD)(pLastPacket - pXmaData) * 8 + nFrameBitOffset;
585 }
