93  void PrepareRendomSamplingTexture()
94  {
95  // Sample kernel
96  std::uniform_real_distribution<GLfloat> randomFloatsClamped(0.2, 1.0); // generates random floats between 0.0 and 1.0
97  std::uniform_real_distribution<GLfloat> randomFloats(0.0, 1.0); // generates random floats between 0.0 and 1.0
98  std::default_random_engine generator;
99  for (GLuint i = 0; i < 32; ++i)
100  {
101  glm::vec3 sample(randomFloats(generator) * 2.0 - 1.0, randomFloats(generator) * 2.0 - 1.0, randomFloatsClamped(generator));
102  sample = glm::normalize(sample);
103  sample *= randomFloats(generator);
104  GLfloat scale = GLfloat(i) / 32.0;
105 
106  // Scale samples s.t. they're more aligned to center of kernel
107  scale = Helpers::lerp(0.1f, 1.0f, scale * scale);
108  sample *= scale;
109  ssaoKernel.push_back(sample);
110  }
111 
112  // Noise texture
113  for (GLuint i = 0; i < 64; i++)
114  {
115  glm::vec3 noise(randomFloats(generator) * 2.0 - 1.0, randomFloats(generator) * 2.0 - 1.0, 0.0f); // rotate around z-axis (in tangent space)
116  ssaoNoise.push_back(noise);
117  }
