178 HRESULT XAudio2Create(__deref_out IXAudio2** ppXAudio2, UINT32 Flags X2DEFAULT(0),
179  XAUDIO2_PROCESSOR XAudio2Processor X2DEFAULT(XAUDIO2_DEFAULT_PROCESSOR))
180 {
181  // Instantiate the appropriate XAudio2 engine
182  IXAudio2* pXAudio2;
183  HRESULT hr = CoCreateInstance((Flags & XAUDIO2_DEBUG_ENGINE) ? CLSID_XAudio2_Debug : CLSID_XAudio2,
184  NULL, CLSCTX_INPROC_SERVER, IID_IXAudio2, (void**)&pXAudio2);
185  if (SUCCEEDED(hr))
186  {
187  hr = pXAudio2->Initialize(Flags, XAudio2Processor);
188 
189  if (SUCCEEDED(hr))
190  {
191  *ppXAudio2 = pXAudio2;
192  }
193  else
194  {
195  pXAudio2->Release();
196  }
197  }
198  return hr;
199 }
